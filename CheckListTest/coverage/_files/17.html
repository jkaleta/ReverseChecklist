843        private class ListViewGestureListener extends SimpleOnGestureListener
844        {
845                private final int SWIPE_MAX_OFF_PATH = 300;
846                private final int SWIPE_MIN_DISTANCE = 150;
847                private final int SWIPE_THRESHOLD_VELOCITY = 100;
848                private final String TAG = this.getClass().getName();
849 
850                @Override
851                public boolean onFling(MotionEvent e1, MotionEvent e2, float velocityX, float velocityY)
852                {
853                        try
854                        {
855                                if (Math.abs(e1.getY() - e2.getY()) > SWIPE_MAX_OFF_PATH)
856                                        return false;
857                                // right to left swipe
858                                if (e1.getX() - e2.getX() > SWIPE_MIN_DISTANCE
859                                                && Math.abs(velocityX) > SWIPE_THRESHOLD_VELOCITY)
860                                {
861                                        flipper.showNext();
862                                        setCurrentCategoryTitleAndInfo(true);
863                                        return true;
864 
865                                }
866                                else if (e2.getX() - e1.getX() > SWIPE_MIN_DISTANCE
867                                                && Math.abs(velocityX) > SWIPE_THRESHOLD_VELOCITY)
868                                {
869                                        flipper.showPrevious();
870                                        setCurrentCategoryTitleAndInfo(true);
871                                        return true;
872                                }
873                        }
874                        catch (Exception e)
875                        {
876                                Log.e(TAG, "An exception occurred trying to flip to next tab.");
877                        }
878                        return false;
879                }
880        }
